1. 配置为softap模式
rmmod xradio_wlan.ko
rmmod xradio_core.ko
rmmod xradio_mac.ko
insmod xradio_mac.ko
insmod xradio_core.ko
insmod xradio_wlan.ko
ifconfig wlan0 up
hostapd -d /etc/wifi/hostapd.conf &
ifconfig wlan0 192.168.3.25 netmask 255.255.255.0
udhcpd /etc/wifi/udhcpd.conf

ifconfig wlan0 192.168.3.25 netmask 255.255.255.0

2. 切换为station模式
ifconfig mon.wlan0 down
rmmod xradio_wlan.ko
rmmod xradio_core.ko
rmmod xradio_mac.ko
insmod xradio_mac.ko
insmod xradio_core.ko
insmod xradio_wlan.ko
ifconfig wlan0 up
udhcpc -b -i wlan0 R

wpa_supplicant -i wlan0 -c /etc/wifi/wpa_supplicant.conf -B
